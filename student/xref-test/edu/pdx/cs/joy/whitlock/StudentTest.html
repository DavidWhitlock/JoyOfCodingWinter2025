84    @Test
85    void aboutDatesKoanUsingDateTimeFormatterToFormatDate() {
86      LocalDateTime date = LocalDateTime.ofInstant(Instant.ofEpochMilli(100010001000L), ZoneId.of("-07:00"));
87  
88      // assert with and without non-breaking space to handle Java 17 and 21+
89  
90      String shortFormat = DateTimeFormatter.ofLocalizedDateTime(FormatStyle.SHORT).format(date);
91      assertThat(shortFormat, anyOf(equalTo("3/3/73, 5:33\u202FAM"), equalTo("3/3/73, 5:33 AM")));
92  
93      String mediumFormat = DateTimeFormatter.ofLocalizedDateTime(FormatStyle.MEDIUM).format(date);
94      assertThat(mediumFormat, anyOf(equalTo("Mar 3, 1973, 5:33:21\u202FAM"), equalTo("Mar 3, 1973, 5:33:21 AM")));
95    }
